const loadNews = async (query) => {
  setLoading(true);
  setError(null);

    const result = await fetchNews(query, 20);
    if (result.success) {
      setArticles(result.articles);
    } else {
      setError(result.error);
    }
  } catch (err) {
    setError('Terjadi kesalahan...');
  } finally {
    setLoading(false);
  }
